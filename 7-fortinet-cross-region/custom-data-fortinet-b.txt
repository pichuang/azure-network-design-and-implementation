config system global
    set admintimeout 480
    set timezone Asia/Taipei
end

config system probe-response
    set mode http-probe
    set port 8008
    set http-probe-value "OK"
end

config system interface
    edit "port1"
        set allowaccess probe-response ping https
    next
    edit "port2"
        set allowaccess probe-response ping
    next
end

config firewall service custom
    edit "TCP_8008"
        set protocol TCP
        set tcp-portrange 8008
    next
end

config firewall local-in-policy
    edit 1
        set intf "port1"
        set srcaddr "all"
        set dstaddr "all"
        set action accept
        set schedule "always"
        set service "TCP_8008"
    next
end

config system vxlan
  edit vx1
    set interface port1
    set vni 5566
    set remote-ip 172.16.0.4
  next
end

config system interface
    edit "port1"
        set vdom "root"
        set ip 172.26.0.4 255.255.255.192
        set allowaccess ping https probe-response
        set type physical
        set description "external"
        set snmp-index 1
    next
    edit "port2"
        set vdom "root"
        set ip 172.26.0.68 255.255.255.192
        set allowaccess ping probe-response
        set type physical
        set description "internal"
        set snmp-index 2
    next
    edit "vx1"
        set vdom "root"
        set ip 192.168.100.2 255.255.255.0
        set allowaccess ping
        set type vxlan
        set description "Overlay BGP Peering"
        set interface "port1"
    next
end

config router route-map
    edit "changepath"
        config rule
            edit 1
                set match-as-path "ASRpath"
                set set-aspath-action replace
                set set-aspath "10000"
                unset set-ip-prefsrc
            next
            edit 2
                set match-as-path "anypath"
                unset set-ip-prefsrc
            next
        end
    next
end
